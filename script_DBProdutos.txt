CREATE DATABASE DBPRODUTOS;
USE DBPRODUTOS;

CREATE TABLE PRODUTOS(
CODIGO INT NOT NULL AUTO_INCREMENT,
DESCRICAO VARCHAR(100),
PVENDA DECIMAL(9,2),
PCOMPRA DECIMAL(9,2),
ESTOQUEATUAL DECIMAL(9,2),
PRIMARY KEY(CODIGO));
